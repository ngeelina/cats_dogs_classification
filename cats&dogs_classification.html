def read_images(directory, resize_to=(128, 128)):
    """   
    :param directory directory from which to read the files
    :param resize_to a tuple of width, height to resize the images
    """
    files = glob.glob(directory + "*.jpg")
    images = []
    labels = []
    for f in files:
        im = Image.open(f)
        im = im.resize(resize_to)
        im = np.array(im) / 255.0
        im = im.astype("float32")
        images.append(im)
       
        label = 1 if "dog" in f.lower() else 0
        labels.append(label)
       
    return np.array(images), np.array(labels)

def plot_images(images, labels):
    n_cols = min(5, len(images))
    n_rows = len(images) // n_cols
    fig = plt.figure(figsize=(8, 8))
 
    for i in range(n_rows * n_cols):
        sp = fig.add_subplot(n_rows, n_cols, i+1)
        plt.axis("off")
        plt.imshow(images[i])
        sp.set_title(labels[i])
    plt.show()

def num_to_str_labels(labels):
    """
    :param labels numpy array of int
    :returns numpy array of human friendly labels
    """
    str_label =  np.where(labels == 1, "dog", "cat")
    return str_label
 
plot_images(X_train[:10], num_to_str_labels(y_train[:10]))

Building a model¶

The model is built using Keras Functional API. The input layer takes an image of input shape (height x width x channel). Here, the number of channel is 3 as the images are RGB images. The output of this input layer is given as input to the convolutional layer. RELU is used as an activation function to converge faster. Batch normalization normalizes the output of previous layer by subtracting batch mean by batch standard deviation. Max pooling layer reduces the spatial dimension. Flatten layer flattens 3 dimensional data to 1 dimension. Dense layer/ Fully connected layer expects input as a 1D array and has 64 units. The final Dense layer has only 1 unit as this is a binary classification with 2 classes. Sigmoid is used as the activation function as it is between 0 and 1.

The model is defined and inputs and outputs are specified. 'adam' is used as the optimizer and 'binary-crossentropy' is used as a loss function. 'accuracy' is used as the evaluation metric.

In [11]:
image_input = Input(shape=(IM_HEIGHT, IM_WIDTH, 3))
x = Conv2D(filters=32, kernel_size=7, activation = 'relu')(image_input)
x = BatchNormalization()(x)
x = MaxPool2D()(x)
 
x = Conv2D(filters=64, kernel_size=3, activation = 'relu')(x)
x = BatchNormalization()(x)
x = MaxPool2D()(x)
 
x = Conv2D(filters=128, kernel_size=3, activation = 'relu')(x)
x = BatchNormalization()(x)
x = MaxPool2D()(x)

x = Conv2D(filters=256, kernel_size=3, activation = 'relu')(x)
x = BatchNormalization()(x)
x = MaxPool2D()(x)
 
x = Flatten()(x)
x = Dense(units=64, activation = 'relu')(x)
x = BatchNormalization()(x)
x = Dense(units=1)(x)
x = Activation("sigmoid")(x)
 
model = Model(inputs=image_input, outputs=x)
model.compile(optimizer="adam", loss="binary_crossentropy", metrics=["accuracy"])
model.summary()
